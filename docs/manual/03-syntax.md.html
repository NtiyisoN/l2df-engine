Frame example

<frame> 1 test
  pic: 10 wait: 5 next: 999
  state: 4050 hit_a: 330 mp: 55
</frame>

Scene example

<scene>
    <image>
        sprites: { res: "resources/sprites/UI/CS0.png" w: 1280 h: 720 }
        scaleX: 0.5 scaleY: 0.5
    </image>
    <object>
        body: { x: 0 y: 0 z: 0 w: 1024 h: 32 l: 1024 }
        static: true
    </object>
    <object>
        body: { x: 0 y: -1024 z: 0 w: 1024 h: 1184 l: 170 }
        static: true
    </object>
</scene>
